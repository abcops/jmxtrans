49  	@Test
50  	public void canCreateBasicResultData() throws MalformedObjectNameException, InstanceNotFoundException {
51  		Attribute integerAttribute = new Attribute("StartTime", 51L);
52  		ObjectInstance runtime = getRuntime();
53  		List<Result> results = new JmxResultProcessor(
54  				query,
55  				runtime,
56  				ImmutableList.of(integerAttribute),
57  				runtime.getClassName(),
58  				TEST_DOMAIN_NAME).getResults();
59  
60  		assertThat(results).hasSize(1);
61  		Result integerResult = results.get(0);
62  
63  		assertThat(integerResult.getAttributeName()).isEqualTo("StartTime");
64  		assertThat(integerResult.getClassName()).isEqualTo("sun.management.RuntimeImpl");
65  		assertThat(integerResult.getKeyAlias()).isEqualTo("resultAlias");
66  		assertThat(integerResult.getTypeName()).isEqualTo("type=Runtime");
67  		assertThat(integerResult.getValues()).hasSize(1);
68  	}

118 	@Test
119 	public void canReadTabularData() throws MalformedObjectNameException, AttributeNotFoundException, MBeanException,
120 			ReflectionException, InstanceNotFoundException {
121 		ObjectInstance runtime = getRuntime();
122 		AttributeList attr = ManagementFactory.getPlatformMBeanServer().getAttributes(
123 				runtime.getObjectName(), new String[]{"SystemProperties"});
124 		List<Result> results = new JmxResultProcessor(
125 				query,
126 				runtime,
127 				attr.asList(),
128 				runtime.getClassName(),
129 				TEST_DOMAIN_NAME).getResults();
130 
131 		assertThat(results.size()).isGreaterThan(2);
132 
133 		Optional<Result> result = from(results).firstMatch(new ByAttributeName("SystemProperties.java.version"));
134 
135 		assertThat(result.isPresent()).isTrue();
136 
137 		assertThat(result.get().getAttributeName()).isEqualTo("SystemProperties.java.version");
138 		Map<String,Object> values = result.get().getValues();
139 		assertThat(values).hasSize(2);
140 
141 		Object objectValue = result.get().getValues().get("key");
142 		assertThat(objectValue).isInstanceOf(String.class);
143 
144 		String key = (String) objectValue;
145 		assertThat(key).isEqualTo("java.version");
146 	}

223 	private static class ByAttributeName implements Predicate<Result> {
224 		private final String attributeName;
225 
226 		public ByAttributeName(String attributeName) {
227 			this.attributeName = attributeName;
228 		}
229 
230 		@Override
231 		public boolean apply(@Nullable Result result) {
232 			if (result == null) {
233 				return false;
234 			}
235 			return attributeName.equals(result.getAttributeName());
236 		}
