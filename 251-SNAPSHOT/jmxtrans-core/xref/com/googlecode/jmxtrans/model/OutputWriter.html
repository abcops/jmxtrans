12  /**
13   * Interface which defines a writer for taking jmx data and writing it out in
14   * whatever format you want.
15   * 
16   * Note that this class uses a feature of Jackson to serialize anything that
17   * implements this as a "@class". That way, when Jackson deserializes
18   * implementations of this interface, it is done with new objects that implement
19   * this interface.
20   * 
21   * @author jon
22   */
23  @JsonSerialize(include = NON_NULL)
24  @JsonTypeInfo(use = JsonTypeInfo.Id.CLASS, include = JsonTypeInfo.As.PROPERTY, property = "@class")
25  public interface OutputWriter {
26  
27  	void start() throws LifecycleException;
28  
29  	void stop() throws LifecycleException;
30  
31  	void doWrite(Server server, Query query, ImmutableList<Result> results) throws Exception;
32  
33  	/**
34  	 * Settings allow you to configure your Writers with whatever they might
35  	 * need.
36  	 */
37  	Map<String, Object> getSettings();
38  
39  	/**
40  	 * Settings allow you to configure your Writers with whatever they might
41  	 * need.
42  	 */
43  	void setSettings(Map<String, Object> settings);
44  
45  	/**
46  	 * This is run when the object is instantiated. You want to get the settings
47  	 * and validate them.
48  	 */
49  	void validateSetup(Server server, Query query) throws ValidationException;
50  
51  }
