13  /**
14   * Basic filter good for testing that just outputs the Result objects using
15   * System.out.
16   * 
17   * @author jon
18   */
19  public class StdOutWriter extends BaseOutputWriter {
20  
21  	@JsonCreator
22  	public StdOutWriter(
23  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
24  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
25  			@JsonProperty("debug") Boolean debugEnabled,
26  			@JsonProperty("settings") Map<String, Object> settings) {
27  		super(typeNames, booleanAsNumber, debugEnabled, settings);
28  	}
29  
30  	/**
31  	 * nothing to validate
32  	 */
33  	public void validateSetup(Server server, Query query) throws ValidationException {
34  	}
35  
36  	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
37  		for (Result r : results) {
38  			System.out.println(r);
39  		}
40  	}
41  }
