13  import java.util.Map;

15  /**
16   * Naming strategy which uses an JEXL expression to format the name from the result, its class name, attribute name,
17   * and more.  Note the resulting name can be used for any of a number of purposes.  For example, with OpenTSDB it could
18   * be the metric name or a tag name.
19   * <p/>
20   * In the expression, all of the following may be used:
21   * <b>Variables</b>
22   * <dl>
23   * <dt>alias</dt>
24   * <dd>the classname alias of the MBean.</dd>
25   * <dt>attribute</dt>
26   * <dd>the name of the attribute of the MBean queried.</dd>
27   * <dt>class</dt>
28   * <dd>the effective classname of the MBean (uses #alias is not null, otherwise uses #realclass).</dd>
29   * <dt>realclass</dt>
30   * <dd>the class name of the MBean.</dd>
31   * <dt>result</dt>
32   * <dd>the full Result object.</dd>
33   * </dl>
34   */
35  public class JexlNamingStrategy implements NamingStrategy {
36  	private static final Logger LOG = LoggerFactory.getLogger(JexlNamingStrategy.class);
37  
38  	public static final String DEFAULT_EXPRESSION = "class + \".\" + attribute";
39  	public static final String VAR__CLASSNAME = "realclass";
40  	public static final String VAR__ATTRIBUTE_NAME = "attribute";
41  	public static final String VAR__CLASSNAME_ALIAS = "alias";
42  	public static final String VAR__EFFECTIVE_CLASSNAME = "class";
43  	public static final String VAR__TYPENAME = "typename";
44  	public static final String VAR__RESULT = "result";
45  
46  	protected JexlEngine jexl;
47  	protected Expression parsedExpr;
48  
49  
50  	/**
51  	 * Create a new naming strategy using an JEXL expression and the default expression.
52  	 */
53  	public JexlNamingStrategy() throws JexlException {
54  		jexl = new JexlEngine();
55  		this.parsedExpr = jexl.createExpression(DEFAULT_EXPRESSION);
56  	}
57  
58  	/**
59  	 * Create a new naming strategy using an JEXL expression and the given expression.
60  	 *
61  	 * @param expr - the JEXL expression to use to create names.
62  	 */
63  	public JexlNamingStrategy(String expr) throws JexlException {
64  		jexl = new JexlEngine();
65  		this.parsedExpr = jexl.createExpression(expr);
66  	}
67  
68  	/**
69  	 * Format the name for the given result.
70  	 *
71  	 * @param result - the result of a JMX query.
72  	 * @return String - the formatted string resulting from the expression, or null if the formatting fails.
73  	 */
74  	@Override
75  	public String formatName(Result result) {
76  		String formatted;
77  		JexlContext context = new MapContext();
78  
79  		this.populateContext(context, result);
80  		try {
81  			formatted = (String) this.parsedExpr.evaluate(context);
82  		} catch (JexlException jexlExc) {
83  			LOG.error("error applying JEXL expression to query results", jexlExc);
84  			formatted = null;
85  		}
86  
87  		return formatted;
88  	}
89  
90  	public void setExpression(String expr) throws JexlException {
91  		this.parsedExpr = this.jexl.createExpression(expr);
92  	}
93  
94  	/**
95  	 * Populate the context with values from the result.
96  	 *
97  	 * @param context - the expression context used when evaluating JEXL expressions.
98  	 * @param result  - the result of a JMX query.
99  	 */
100 	protected void populateContext(JexlContext context, Result result) {
101 		context.set(VAR__CLASSNAME, result.getClassName());
102 		context.set(VAR__ATTRIBUTE_NAME, result.getAttributeName());
103 		context.set(VAR__CLASSNAME_ALIAS, result.getKeyAlias());
104 
105 		Map<String, String> typeNameMap = KeyUtils.getTypeNameValueMap(result.getTypeName());
106 		context.set(VAR__TYPENAME, typeNameMap);
107 
108 		String effectiveClassname = result.getKeyAlias();
109 		if (effectiveClassname == null) {
110 			effectiveClassname = result.getClassName();
111 		}
112 		context.set(VAR__EFFECTIVE_CLASSNAME, effectiveClassname);
113 
114 		context.set(VAR__RESULT, result);
115 	}
116 }
