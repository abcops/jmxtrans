1   package com.googlecode.jmxtrans.cli;
2   
3   import com.google.common.collect.ImmutableList;
4   import org.apache.commons.cli.CommandLine;
5   import org.apache.commons.cli.CommandLineParser;
6   import org.apache.commons.cli.GnuParser;
7   import org.apache.commons.cli.HelpFormatter;
8   import org.apache.commons.cli.Option;
9   import org.apache.commons.cli.OptionBuilder;
10  import org.apache.commons.cli.Options;
11  
12  import java.io.File;
13  
14  public class CliArgumentParser {
15  	/**
16  	 * Parse the options given on the command line.
17  	 *
18  	 * @param args
19  	 */
20  	public JmxTransConfiguration parseOptions(String[] args) throws OptionsException, org.apache.commons.cli.ParseException {
21  		CommandLineParser parser = new GnuParser();
22  		CommandLine cl = parser.parse(getOptions(), args);
23  		Option[] options = cl.getOptions();
24  
25  		JmxTransConfiguration configuration = new JmxTransConfiguration();
26  
27  		for (Option option : options) {
28  			if (option.getOpt().equals("c")) {
29  				configuration.setContinueOnJsonError(Boolean.parseBoolean(option.getValue()));
30  			} else if (option.getOpt().equals("j")) {
31  				File jsonDir = new File(option.getValue());
32  				if (jsonDir.exists() && jsonDir.isDirectory()) {
33  					configuration.setJsonDirOrFile(jsonDir);
34  				} else {
35  					throw new OptionsException("Path to json directory is invalid: " + jsonDir);
36  				}
37  			} else if (option.getOpt().equals("f")) {
38  				File jsonFile = new File(option.getValue());
39  				if (jsonFile.exists() && jsonFile.isFile()) {
40  					configuration.setJsonDirOrFile(jsonFile);
41  				} else {
42  					throw new OptionsException("Path to json file is invalid: " + jsonFile);
43  				}
44  			} else if (option.getOpt().equals("e")) {
45  				configuration.setRunEndlessly(true);
46  			} else if (option.getOpt().equals("q")) {
47  				File quartzConfigFile = new File(option.getValue());
48  				if (quartzConfigFile.exists() && quartzConfigFile.isFile()) {
49  					configuration.setQuartPropertiesFile(option.getValue());
50  				} else {
51  					throw new OptionsException("Could not find path to the quartz properties file: "
52  							+ quartzConfigFile.getAbsolutePath());
53  				}
54  			} else if (option.getOpt().equals("s")) {
55  				try {
56  					configuration.setRunPeriod(Integer.parseInt(option.getValue()));
57  				} catch (NumberFormatException nfe) {
58  					throw new OptionsException("Seconds between server job runs must be an integer");
59  				}
60  			} else if (option.getOpt().equals("a")) {
61  				ImmutableList.Builder<File> jars = ImmutableList.builder();
62  				for (String jar : option.getValues()) {
63  					jars.add(new File(jar));
64  				}
65  				configuration.setAdditionalJars(jars.build());
66  			} else if (option.getOpt().equals("h")) {
67  				HelpFormatter formatter = new HelpFormatter();
68  				formatter.printHelp("java -jar jmxtrans-all.jar", getOptions());
69  				configuration.setHelp(true);
70  			}
71  		}
72  		if ((!configuration.isHelp()) && (configuration.getJsonDirOrFile() == null)) {
73  			throw new OptionsException("Please specify either the -f or -j option.");
74  		}
75  		return configuration;
76  	}
77  
78  	private Options getOptions() {
79  		Options options = new Options();
80  		options.addOption("c", true, "Continue processing even if one of the JSON configuration file is invalid.");
81  		options.addOption("j", true, "Directory where json configuration is stored. Default is .");
82  		options.addOption("f", true, "A single json file to execute.");
83  		options.addOption("e", false, "Run endlessly. Default false.");
84  		options.addOption("q", true, "Path to quartz configuration file.");
85  		options.addOption("s", true, "Seconds between server job runs (not defined with cron). Default: 60");
86  		options.addOption(OptionBuilder
87  				.withArgName("a")
88  				.withLongOpt("additionalJars")
89  				.hasArgs()
90  				.withValueSeparator(',')
91  				.withDescription("Coma delimited list of additional jars to add to the class path")
92  				.create("a"));
93  		options.addOption("h", false, "Help");
94  		return options;
95  	}
96  }
