28  
29  	private final Logger logger;
30  
31  	@JsonCreator
32  	public Slf4JOutputWriter(
33  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
34  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
35  			@JsonProperty("debug") Boolean debugEnabled,
36  			@JsonProperty("logger") String logger,
37  			@JsonProperty("settings") Map<String, Object> settings) {
38  		super(typeNames, booleanAsNumber, debugEnabled, settings);
39  		String loggerName = MoreObjects.firstNonNull(logger, "jmxtrans.output");
40  		this.logger = LoggerFactory.getLogger(loggerName);
41  	}
42  
43  	@VisibleForTesting
44  	Slf4JOutputWriter(Logger logger) {
45  		super(ImmutableList.<String>of(),
46  				true,
47  				false,
48  				new HashMap<String, Object>());
49  		this.logger = logger;
50  	}
51  
52  	@Override
53  	protected void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
54  		final List<String> typeNames = getTypeNames();
55  
56  		for (final Result result : results) {
57  			final Map<String, Object> resultValues = result.getValues();
58  			if (resultValues != null) {
59  				for (final Map.Entry<String, Object> values : resultValues.entrySet()) {
60  					logValue(server, query, typeNames, result, values);
61  				}
62  			}
63  		}
64  	}
65  
66  	private void logValue(Server server, Query query, List<String> typeNames, Result result, Map.Entry<String, Object> values) throws IOException {
67  		Object value = values.getValue();
68  
69  		if (value != null && isNumeric(value)) {
70  			Closer closer = Closer.create();
71  			try {
72  				closer.register(MDC.putCloseable("server", computeAlias(server)));
73  				closer.register(MDC.putCloseable("metric", KeyUtils.getKeyString(server, query, result, values, typeNames, null)));
74  				closer.register(MDC.putCloseable("value", value.toString()));
75  				if (result.getKeyAlias() != null) {
76  					closer.register(MDC.putCloseable("resultAlias", result.getKeyAlias()));
77  				}
78  				closer.register(MDC.putCloseable("attributeName", result.getAttributeName()));
79  				closer.register(MDC.putCloseable("key", values.getKey()));
80  				closer.register(MDC.putCloseable("epoch", valueOf(result.getEpoch())));
81  
82  				logger.info("");
83  			} catch (Throwable t) {
84  				closer.rethrow(t);
85  			} finally {
86  				closer.close();
87  			}
88  		}
89  	}
90  
91  	private String computeAlias(Server server) {
92  		if (server.getAlias() != null) return server.getAlias();
93  		return cleanupStr(server.getHost() + "_" + server.getPort());
94  	}
95  
96  	@Override
97  	public void validateSetup(Server server, Query query) throws ValidationException {}
