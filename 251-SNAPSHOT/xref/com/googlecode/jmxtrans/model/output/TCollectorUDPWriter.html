22  /**
23   * {@link com.googlecode.jmxtrans.model.OutputWriter} for the <a href="https://github.com/OpenTSDB/tcollector/blob/master/collectors/0/udp_bridge.py">TCollector udp_bridge</a>.
24   * Largely based on StatsDWriter and OpenTSDBWriter
25   *
26   * @author Kieren Hynd
27   * @author Arthur Naseef
28   */
29  public class TCollectorUDPWriter extends OpenTSDBGenericWriter {
30  	private static final Logger log = LoggerFactory.getLogger(TCollectorUDPWriter.class);
31  
32  	protected SocketAddress address;
33  	protected DatagramSocket dgSocket;
34  
35  	@JsonCreator
36  	public TCollectorUDPWriter(
37  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
38  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
39  			@JsonProperty("debug") Boolean debugEnabled,
40  			@JsonProperty("host") String host,
41  			@JsonProperty("port") Integer port,
42  			@JsonProperty("tags") Map<String, String> tags,
43  			@JsonProperty("tagName") String tagName,
44  			@JsonProperty("mergeTypeNamesTags") Boolean mergeTypeNamesTags,
45  			@JsonProperty("metricNamingExpression") String metricNamingExpression,
46  			@JsonProperty("addHostnameTag") Boolean addHostnameTag,
47  			@JsonProperty("settings") Map<String, Object> settings) throws LifecycleException, UnknownHostException {
48  		super(typeNames, booleanAsNumber, debugEnabled, host, port, tags, tagName, mergeTypeNamesTags, metricNamingExpression,
49  				addHostnameTag, settings);
50  	}
51  
52  	/**
53  	 * Do not add the hostname tag "host" with the name of the host by default since tcollector normally adds the
54  	 * hostname.
55  	 */
56  	@Override
57  	protected boolean getAddHostnameTagDefault() {
58  		return false;
59  	}
60  
61  	/**
62  	 * Setup at start of the writer.
63  	 */
64  	@Override
65  	public void prepareSender() throws LifecycleException {
66  
67  		if (host == null || port == null) {
68  			throw new LifecycleException("Host and port for " + this.getClass().getSimpleName() + " output can't be null");
69  		}
70  
71  		try {
72  			this.dgSocket = new DatagramSocket();
73  			this.address = new InetSocketAddress(host, port);
74  		} catch (SocketException sockExc) {
75  			log.error("Failed to create a datagram socket", sockExc);
76  			throw new LifecycleException(sockExc);
77  		}
78  	}
79  
80  	/**
81  	 * Send a single metric to TCollector.
82  	 *
83  	 * @param metricLine - the line containing the metric name, value, and tags for a single metric; excludes the
84  	 *                   "put" keyword expected by OpenTSDB and the trailing newline character.
85  	 */
86  	@Override
87  	protected void sendOutput(String metricLine) throws IOException {
88  		DatagramPacket packet;
89  		byte[] data;
90  
91  		data = metricLine.getBytes("UTF-8");
92  		packet = new DatagramPacket(data, 0, data.length, this.address);
93  
94  		this.dgSocket.send(packet);
95  	}
