1   package com.googlecode.jmxtrans.model.output;

3   import com.fasterxml.jackson.annotation.JsonCreator;
4   import com.fasterxml.jackson.annotation.JsonProperty;
5   import com.google.common.collect.ImmutableList;
6   import org.apache.log4j.Appender;
7   import org.apache.log4j.DailyRollingFileAppender;
8   import org.apache.log4j.PatternLayout;
9   
10  import java.io.IOException;
11  import java.util.Map;
12  
13  /**
14   * This class is derived from KeyOutWriter. It uses DailyRollingFileAppender
15   * instead of RollingFileAppender.
16   * 
17   * Writes out data in the same format as the GraphiteWriter, except to a file
18   * and tab delimited. Takes advantage of Log4J DailyRollingFileAppender to
19   * automatically handle rolling the files at defined intervals (datePattern).
20   * 
21   * Note that this writer will NOT clean up after itself. The files will not be
22   * deleted. It is up to the user to clean things up.
23   * 
24   * The datePattern is taken directly from DailyRollingFileAppender DatePattern.
25   * 
26   * The default datePattern will roll the files at midnight once a day
27   * ("'.'yyyy-MM-dd") See the documentation for DailyRollingFileAppender for
28   * other useful patterns.
29   * 
30   */
31  public class DailyKeyOutWriter extends KeyOutWriter {
32  
33  	private static final String DATE_PATTERN = "'.'yyyy-MM-dd";
34  
35  	private final String datePattern;
36  
37  	@JsonCreator
38  	public DailyKeyOutWriter(
39  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
40  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
41  			@JsonProperty("debug") Boolean debugEnabled,
42  			@JsonProperty("outputFile") String outputFile,
43  			@JsonProperty("maxLogFileSize") String maxLogFileSize,
44  			@JsonProperty("maxLogBackupFiles") int maxLogBackupFiles,
45  			@JsonProperty("delimiter") String delimiter,
46  			@JsonProperty("datePattern") String datePattern,
47  			@JsonProperty("settings") Map<String, Object> settings) {
48  		super(typeNames, booleanAsNumber, debugEnabled, outputFile, maxLogFileSize, maxLogBackupFiles, delimiter, settings);
49  		this.datePattern = firstNonNull(
50  				datePattern,
51  				(String) getSettings().get("datePattern"),
52  				DATE_PATTERN
53  		);
54  	}
55  
56  	/**
57  	 * The maxLogFileSize and maxLogBackupFiles are ignored as per the existing behaviour of DailyKeyOutWriter.
58  	 */
59  	@Override
60  	protected Appender buildLog4jAppender(String fileStr, String maxLogFileSize, Integer maxLogBackupFiles)
61  			throws IOException {
62  		return new DailyRollingFileAppender(new PatternLayout(LOG_PATTERN), fileStr, datePattern);
63  	}
64  	
65  	@Override
66  	protected String buildLoggerName() {
67  		return "DailyKeyOutWriter" + this.hashCode();
68  	}
69  
70  	public String getDatePattern() {
71  		return datePattern;
72  	}
73  }
