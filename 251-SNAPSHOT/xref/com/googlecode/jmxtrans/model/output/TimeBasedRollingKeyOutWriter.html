13  import com.fasterxml.jackson.annotation.JsonProperty;
14  import com.google.common.collect.ImmutableList;
15  import org.slf4j.Logger;
16  
17  import java.io.IOException;
18  import java.util.Map;

20  /**
21   * Extension of KeyOutWriter to use Logback for its logging.  This version supports outputting with a time and size
22   * based rollover policy but with no file rename taking place.  The filename remains constant for the output file from
23   * its creation to the roll over to a new file.  Then the old file remains with its old name and the new file is used
24   * for further logging.
25   * 
26   * We found this to be useful when dealing with forwarding logs on Windows as our other support applications were not
27   * able to handle the rename correctly on Windows and DailyKeyOutWriter caused the original file to be renamed.  
28   * Also DailyRollingAppender has bugs that can cause data loss and it is no longer recommended for use judging by the
29   * javadocs for it.
30   * 
31   * The basic operation is to set the date time to rollover in the outputFile setting.  For example:
32   *      "outputFile": "testing-here.log%d{yyyy-MM-dd}.%i",
33   * In this case we have set the logging to roll over every day.  There is also a default max size as per the
34   * KeyOutWriter.  If the file exceeds the default max the .%i integer is increments.  See the Logstash documentation
35   * for more information.
36   */
37  public class TimeBasedRollingKeyOutWriter extends KeyOutWriter {
38  	
39  	private static final LoggerContext loggerContext = new LoggerContext();
40  	private static final String DEFAULT_OUTPUT_PATTERN = "%msg%n";
41  	private static final String SETTING_OUTPUT_PATTERN = "outputPattern";
42  
43  	@JsonCreator
44  	public TimeBasedRollingKeyOutWriter(
45  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
46  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
47  			@JsonProperty("debug") Boolean debugEnabled,
48  			@JsonProperty("outputFile") String outputFile,
49  			@JsonProperty("maxLogFileSize") String maxLogFileSize,
50  			@JsonProperty("maxLogBackupFiles") int maxLogBackupFiles,
51  			@JsonProperty("delimiter") String delimiter,
52  			@JsonProperty("settings") Map<String, Object> settings) {
53  		super(typeNames, booleanAsNumber, debugEnabled, outputFile, maxLogFileSize, maxLogBackupFiles, delimiter, settings);
54  	}
55  
56  	@Override
57  	protected Logger initLogger(String fileStr) throws IOException {
58  		RollingPolicy rollingPolicy = initRollingPolicy(fileStr, getMaxLogBackupFiles(), getMaxLogFileSize());
59  		RollingFileAppender appender = buildAppender(buildEncoder(), rollingPolicy);
60  		
61  		rollingPolicy.start();
62  		appender.start();
63  		
64  		// configure the logger for info and add the appender
65  		return getAndConfigureLogger(appender);
66  	}
67  	
68  	private RollingFileAppender buildAppender(Encoder encoder, RollingPolicy rollingPolicy) {
69  		RollingFileAppender appender = new RollingFileAppender();
70  		appender.setEncoder(encoder);
71  		appender.setAppend(true);
72  		appender.setContext(loggerContext);
73  		appender.setRollingPolicy(rollingPolicy);
74  		
75  		rollingPolicy.setParent(appender);
76  		return appender;
77  	}
78  	
79  	protected Logger getAndConfigureLogger(Appender appender) {
80  		ch.qos.logback.classic.Logger loggerLocal = loggerContext.getLogger(buildLoggerName());
81  		loggerLocal.setAdditive(false);
82  		loggerLocal.setLevel(Level.INFO);
83  		loggerLocal.addAppender(appender);
84  		return loggerLocal;
85  	}
86  	
87  	protected Encoder buildEncoder() {
88  		PatternLayoutEncoder logEncoder = new PatternLayoutEncoder();
89  		logEncoder.setContext(loggerContext);
90  		logEncoder.setPattern(getSettingOutputPattern());
91  		logEncoder.start();
92  		
93  		return logEncoder;
94  	}
95  				
96  	protected RollingPolicy initRollingPolicy(String fileName, int maxBackupFiles, String maxFileSize) {
97  		SizeAndTimeBasedFNATP sizeTimeBasedPolicy = new SizeAndTimeBasedFNATP();
98  		// the max file size before rolling to a new file
99  		sizeTimeBasedPolicy.setMaxFileSize(maxFileSize);
100 		sizeTimeBasedPolicy.setContext(loggerContext);
101 		
102 		TimeBasedRollingPolicy policy = new TimeBasedRollingPolicy();
103 		// set the filename pattern
104 		policy.setFileNamePattern(fileName);
105 		// the maximum number of backup files to keep around
106 		policy.setMaxHistory(maxBackupFiles);
107 		policy.setTimeBasedFileNamingAndTriggeringPolicy(sizeTimeBasedPolicy);
108 		policy.setContext(loggerContext);
109 		
110 		return policy;
111 	}
112 	
113 	protected String getSettingOutputPattern() {
114 		String outputPattern = (String) this.getSettings().get(SETTING_OUTPUT_PATTERN);
115 		if (outputPattern == null) {
116 			return DEFAULT_OUTPUT_PATTERN;
117 		}
118 		else {
119 			return outputPattern;
120 		}
121 	}
122 }
