25  /**
26   * <a href="http://sensuapp.org/docs/0.12/events">Sensu Event Data</a>
27   * Format from <a href=https://github.com/SimpleFinance/chef-handler-sensu-event">chef-handler-sensu-event</a>
28   * Class structure from LibratoWriter
29   * <p/>
30   * Settings:
31   * <ul>
32   * <li>"{@code host}": Sensu client host. Optional, default value: {@value #DEFAULT_SENSU_HOST}</li>
33   * <li>"{@code handler}": Sensu handler. Optional, default value: {@value #DEFAULT_SENSU_HANDLER}</li>
34   * </ul>
35   *
36   * @author <a href="mailto:jhmartin@toger.us">Jason Martin</a>
37   */
38  public class SensuWriter extends BaseOutputWriter {
39  
40  	private final Logger logger = LoggerFactory.getLogger(getClass());
41  
42  	private final JsonFactory jsonFactory = new JsonFactory();
43  
44  	public final static String SETTING_HANDLER = "handler";
45  	public final static String DEFAULT_SENSU_HOST = "localhost";
46  	public final static String DEFAULT_SENSU_HANDLER = "graphite";
47  
48  	/**
49  	 * Sensu HTTP API URL
50  	 */
51  	private final String host;
52  	private final String handler;
53  
54  	@JsonCreator
55  	public SensuWriter(
56  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
57  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
58  			@JsonProperty("debug") Boolean debugEnabled,
59  			@JsonProperty("host") String host,
60  			@JsonProperty("handler") String handler,
61  			@JsonProperty("settings") Map<String, Object> settings) {
62  		super(typeNames, booleanAsNumber, debugEnabled, settings);
63  		this.host = firstNonNull(host, (String) getSettings().get(HOST), DEFAULT_SENSU_HOST);
64  		this.handler = firstNonNull(handler, (String) getSettings().get(SETTING_HANDLER), DEFAULT_SENSU_HANDLER);
65  	}
66  
67  	public void validateSetup(Server server, Query query) throws ValidationException {
68  		logger.info("Start Sensu writer connected to '{}' with handler {}", host, handler);
69  	}
70  
71  	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
72  		logger.debug("Export to '{}', metrics {}", host, query);
73  		writeToSensu(server, query, results);
74  	}
75  
76  	private void serialize(Server server, Query query, List<Result> results, OutputStream outputStream) throws IOException {
77  		JsonGenerator g = jsonFactory.createGenerator(outputStream, JsonEncoding.UTF8);
78  		g.useDefaultPrettyPrinter();
79  		g.writeStartObject();
80  		g.writeStringField("name", "jmxtrans");
81  		g.writeStringField("type", "metric");
82  		g.writeStringField("handler", handler);
83  
84  		StringBuilder jsonoutput = new StringBuilder();
85  		List<String> typeNames = getTypeNames();
86  		for (Result result : results) {
87  			Map<String, Object> resultValues = result.getValues();
88  			if (resultValues != null) {
89  				for (Map.Entry<String, Object> values : resultValues.entrySet()) {
90  					if (NumberUtils.isNumeric(values.getValue())) {
91  						Object value = values.getValue();
92  						jsonoutput.append(KeyUtils.getKeyString(server, query, result, values, typeNames, null)).append(" ")
93  								.append(value).append(" ")
94  								.append(TimeUnit.SECONDS.convert(result.getEpoch(), TimeUnit.MILLISECONDS))
95  								.append(System.getProperty("line.separator"));
96  					}
97  				}
98  			}
99  		}
100 		g.writeStringField("output", jsonoutput.toString());
101 		g.writeEndObject();
102 		g.flush();
103 		g.close();
104 	}
105 
106 	private void writeToSensu(Server server, Query query, List<Result> results) {
107 		Socket socketConnection = null;
108 		try {
109 			socketConnection = new Socket(host, 3030);
110 			serialize(server, query, results, socketConnection.getOutputStream());
111 		} catch (Exception e) {
112 			logger.warn("Failure to send result to Sensu server '{}'", host, e);
113 		} finally {
114 			if (socketConnection != null) {
115 				try {
116 					socketConnection.close();
117 				} catch (IOException e) {
118 					logger.warn("Exception closing Sensu connection", e);
119 				}
120 			}
121 		}
122 	}
123 
124 	public String getHost() {
125 		return host;
126 	}
127 
128 	public String getHandler() {
129 		return handler;
130 	}
