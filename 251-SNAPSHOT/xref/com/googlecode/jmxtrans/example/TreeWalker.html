20  /**
21   * Walks a JMX tree and prints out all of the attribute values.
22   * 
23   * Does not use the JmxTrans api. This is just more of a proof of concept and
24   * shows how utterly broken and stupid the jmx api is. The fact that you have to
25   * pass in null,null into MBeanServerConnection.queryNames() is utterly stupid.
26   * 
27   * This code was borrowed from the munin jmxquery plugin which I had to jad
28   * decompile since I couldn't find the source to it.
29   * 
30   * @author jon
31   */
32  public class TreeWalker {
33  
34  	private static final Logger log = LoggerFactory.getLogger(TreeWalker.class);
35  
36  	public static void main(String[] args) throws Exception {
37  		Server server = Server.builder().setHost("localhost").setPort("1099").build();
38  
39  		JMXConnector conn = null;
40  		try {
41  			conn = server.getServerConnection();
42  			MBeanServerConnection mbeanServer = conn.getMBeanServerConnection();
43  
44  			TreeWalker tw = new TreeWalker();
45  			tw.walkTree(mbeanServer);
46  		} catch (IOException e) {
47  			log.error("Problem processing queries for server: " + server.getHost() + ":" + server.getPort(), e);
48  		} finally {
49  			if (conn != null) {
50  				conn.close();
51  			}
52  		}
53  	}
54  
55  	public void walkTree(MBeanServerConnection connection) throws Exception {
56  
57  		// key here is null, null returns everything!
58  		Set<ObjectName> mbeans = connection.queryNames(null, null);
59  		for (ObjectName name : mbeans) {
60  			MBeanInfo info = connection.getMBeanInfo(name);
61  			MBeanAttributeInfo[] attrs = info.getAttributes();
62  			String[] attrNames = new String[attrs.length];
63  			for (int i = 0; i < attrs.length; i++) {
64  				attrNames[i] = attrs[i].getName();
65  			}
66  			try {
67  				AttributeList attributes = connection.getAttributes(name, attrNames);
68  				for (Attribute attribute : attributes.asList()) {
69  					output(name.getCanonicalName() + "%" + attribute.getName(), attribute.getValue());
70  				}
71  			} catch (Exception e) {
72  				log.error("error getting " + name + ":" + e.getMessage(), e);
73  			}
74  		}
75  	}
76  
77  	public void output(String name, Object attr) {
78  		CompositeDataSupport cds;
79  		if ((attr instanceof CompositeDataSupport)) {
80  			cds = (CompositeDataSupport) attr;
81  			for (String key : cds.getCompositeType().keySet()) {
82  				log.info(name + "." + key + ".value " + format(cds.get(key)));
83  			}
84  		} else {
85  			log.info(name + ".value " + format(attr));
86  		}
87  	}
88  
89  	public String format(Object value) {
90  		if (value == null) {
91  			return null;
92  		} else if ((value instanceof String)) {
93  			return (String) value;
94  		} else if ((value instanceof Number)) {
95  			NumberFormat f = NumberFormat.getInstance();
96  			f.setMaximumFractionDigits(2);
97  			f.setGroupingUsed(false);
98  			return f.format(value);
99  		} else if ((value instanceof Object[])) {
100 			return Integer.toString(Arrays.asList((Object[]) value).size());
101 		}
102 		return value.toString();
103 	}
104 }
