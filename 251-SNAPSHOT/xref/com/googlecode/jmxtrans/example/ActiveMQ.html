1   package com.googlecode.jmxtrans.example;
2   
3   import com.google.inject.Guice;
4   import com.google.inject.Injector;
5   import com.googlecode.jmxtrans.JmxTransformer;
6   import com.googlecode.jmxtrans.cli.JmxTransConfiguration;
7   import com.googlecode.jmxtrans.guice.JmxTransModule;
8   import com.googlecode.jmxtrans.model.JmxProcess;
9   import com.googlecode.jmxtrans.model.Query;
10  import com.googlecode.jmxtrans.model.Server;
11  import com.googlecode.jmxtrans.model.output.GraphiteWriter;
12  import com.googlecode.jmxtrans.util.JsonPrinter;
13  
14  /**
15   * This example shows how to query an ActiveMQ server for some information.
16   * 
17   * The point of this example is to show that * works as part of the objectName.
18   * It also shows that you don't have to set an attribute to get for a query.
19   * jmxtrans will get all attributes on an object if you don't specify any.
20   * 
21   * @author jon
22   */
23  public class ActiveMQ {
24  
25  	private static final String GW_HOST = "192.168.192.133";
26  
27  	public static void main(String[] args) throws Exception {
28  
29  		Server.Builder serverBuilder = Server.builder()
30  				.setHost("w2")
31  				.setPort("1105")
32  				.setAlias("w2_activemq_1105");
33  
34  		GraphiteWriter gw = GraphiteWriter.builder()
35  				.addTypeName("destinationName")
36  				.addTypeName("Destination")
37  				.setDebugEnabled(true)
38  				.setHost(GW_HOST)
39  				.setPort(2003)
40  				.build();
41  
42  		Query q = Query.builder()
43  				.setObj("org.apache.activemq:BrokerName=localhost,Type=Subscription,clientId=*,consumerId=*,destinationName=*,destinationType=Queue,persistentMode=Non-Durable")
44  				.addAttr("PendingQueueSize")
45  				.addAttr("DispatchedQueueSize")
46  				.addAttr("EnqueueCounter")
47  				.addAttr("DequeueCounter")
48  				.addAttr("MessageCountAwaitingAcknowledge")
49  				.addAttr("DispachedCounter")
50  				.addOutputWriter(gw)
51  				.build();
52  		serverBuilder.addQuery(q);
53  
54  		Query q2 = Query.builder()
55  				.setObj("org.apache.activemq:BrokerName=localhost,Destination=ActiveMQ.Advisory.Consumer.Queue.*,Type=Topic")
56  				.addAttr("QueueSize")
57  				.addAttr("MaxEnqueueTime")
58  				.addAttr("MinEnqueueTime")
59  				.addAttr("AverageEnqueueTime")
60  				.addAttr("InFlightCount")
61  				.addAttr("ConsumerCount")
62  				.addAttr("ProducerCount")
63  				.addAttr("DispatchCount")
64  				.addAttr("DequeueCount")
65  				.addAttr("EnqueueCount")
66  				.addAttr("Subscriptions")
67  				.addOutputWriter(gw)
68  				.build();
69  		serverBuilder.addQuery(q2);
70  
71  		Query q3 = Query.builder()
72  				.setObj("org.apache.activemq:BrokerName=localhost,Destination=*,Type=Queue")
73  				.addAttr("QueueSize")
74  				.addAttr("MaxEnqueueTime")
75  				.addAttr("MinEnqueueTime")
76  				.addAttr("AverageEnqueueTime")
77  				.addAttr("InFlightCount")
78  				.addAttr("ConsumerCount")
79  				.addAttr("ProducerCount")
80  				.addAttr("DispatchCount")
81  				.addAttr("DequeueCount")
82  				.addAttr("EnqueueCount")
83  				.addAttr("Subscriptions")
84  				.addOutputWriter(gw)
85  				.build();
86  		serverBuilder.addQuery(q3);
87  
88  		Query q4 = Query.builder()
89  				.setObj("org.apache.activemq:BrokerName=localhost,Destination=*,Type=Topic")
90  				.addAttr("QueueSize")
91  				.addAttr("MaxEnqueueTime")
92  				.addAttr("MinEnqueueTime")
93  				.addAttr("AverageEnqueueTime")
94  				.addAttr("InFlightCount")
95  				.addAttr("ConsumerCount")
96  				.addAttr("ProducerCount")
97  				.addAttr("DispatchCount")
98  				.addAttr("DequeueCount")
99  				.addAttr("EnqueueCount")
100 				.addAttr("Subscriptions")
101 				.addOutputWriter(gw)
102 				.build();
103 		serverBuilder.addQuery(q4);
104 
105 		Query q5 = Query.builder()
106 				.setObj("org.apache.activemq:BrokerName=localhost,Type=Broker")
107 				.addOutputWriter(gw)
108 				.build();
109 		serverBuilder.addQuery(q5);
110 
111 		Query q6 = Query.builder()
112 				.setObj("java.lang:type=Memory")
113 				.addAttr("HeapMemoryUsage")
114 				.addAttr("NonHeapMemoryUsage")
115 				.addOutputWriter(gw)
116 				.build();
117 		serverBuilder.addQuery(q6);
118 
119 		Query q7 = Query.builder()
120 				.setObj("java.lang:type=Threading")
121 				.addAttr("DaemonThreadCount")
122 				.addAttr("PeakThreadCount")
123 				.addAttr("ThreadCount")
124 				.addAttr("CurrentThreadCpuTime")
125 				.addAttr("CurrentThreadUserTime")
126 				.addAttr("TotalStartedThreadCount")
127 				.addOutputWriter(gw)
128 				.build();
129 		serverBuilder.addQuery(q7);
130 
131 		Query q8 = Query.builder()
132 				.setObj("java.lang:name=*,type=GarbageCollector")
133 				.addKey("committed")
134 				.addKey("init")
135 				.addKey("max")
136 				.addKey("used")
137 				.addKey("duration")
138 				.addKey("CollectionCount")
139 				.addKey("CollectionTime")
140 				.addOutputWriter(gw)
141 				.build();
142 		serverBuilder.addQuery(q8);
143 
144 
145 		Query q9 = Query.builder()
146 				.setObj("java.lang:type=MemoryPool,name=*")
147 				.addOutputWriter(GraphiteWriter.builder()
148 						.addTypeName("name")
149 						.setDebugEnabled(true)
150 						.setHost(GW_HOST)
151 						.setPort(2003)
152 						.build())
153 				.build();
154 		serverBuilder.addQuery(q9);
155 
156 		JmxProcess process = new JmxProcess(serverBuilder.build());
157 		new JsonPrinter(System.out).prettyPrint(process);
158 
159 		Injector injector = Guice.createInjector(new JmxTransModule(new JmxTransConfiguration()));
160 		JmxTransformer transformer = injector.getInstance(JmxTransformer.class);
161 		transformer.executeStandalone(process);
162 
163 		// for (int i = 0; i < 160; i++) {
164 		// JmxUtils.processServer(server);
165 		// Thread.sleep(1000);
166 		// }
167 
168 	}
169 }
