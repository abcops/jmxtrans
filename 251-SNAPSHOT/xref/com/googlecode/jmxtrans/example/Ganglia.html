1   package com.googlecode.jmxtrans.example;
2   
3   import com.google.inject.Guice;
4   import com.google.inject.Injector;
5   import com.googlecode.jmxtrans.JmxTransformer;
6   import com.googlecode.jmxtrans.cli.JmxTransConfiguration;
7   import com.googlecode.jmxtrans.guice.JmxTransModule;
8   import com.googlecode.jmxtrans.model.JmxProcess;
9   import com.googlecode.jmxtrans.model.Query;
10  import com.googlecode.jmxtrans.model.Server;
11  import com.googlecode.jmxtrans.model.output.GangliaWriter;
12  import com.googlecode.jmxtrans.util.JsonPrinter;
13  
14  /**
15   * This class hits a Graphite server running on port 2003 and sends the memory
16   * usage data to it for graphing.
17   * 
18   * @author jon
19   */
20  public class Ganglia {
21  
22  	private static final JsonPrinter printer = new JsonPrinter(System.out);
23  
24  	public static void main(String[] args) throws Exception {
25  		printer.prettyPrint(new JmxProcess(Server.builder()
26  				.setHost("w2")
27  				.setPort("1099")
28  				.setAlias("fooalias")
29  				.addQuery(Query.builder()
30  						.setObj("java.lang:type=GarbageCollector,name=ConcurrentMarkSweep")
31  						.addOutputWriter(GangliaWriter.builder()
32  							.setHost("10.0.3.16")
33  							.setPort(8649)
34  							.setDebugEnabled(true)
35  							.setGroupName("memory")
36  							.build())
37  					.build())
38  				.build()));
39  
40  		Injector injector = Guice.createInjector(new JmxTransModule(new JmxTransConfiguration()));
41  		JmxTransformer transformer = injector.getInstance(JmxTransformer.class);
42  		transformer.executeStandalone(new JmxProcess(Server.builder()
43  				.setHost("w2")
44  				.setPort("1099")
45  				.setAlias("fooalias")
46  				.addQuery(Query.builder()
47  						.setObj("java.lang:type=GarbageCollector,name=ConcurrentMarkSweep")
48  						.addOutputWriter(GangliaWriter.builder()
49  							.setHost("10.0.3.16")
50  							.setPort(8649)
51  							.setDebugEnabled(true)
52  							.setGroupName("memory")
53  							.build())
54  					.build())
55  				.build()));
56  	}
57  
58  }
