3   import java.io.File;
4   import java.io.FileNotFoundException;
5   import java.lang.reflect.InvocationTargetException;
6   import java.lang.reflect.Method;
7   import java.net.MalformedURLException;
8   import java.net.URL;
9   import java.net.URLClassLoader;
10  
11  public class ClassLoaderEnricher {
12  
13  	public void add(File file) throws MalformedURLException, FileNotFoundException {
14  		if (!file.exists()) {
15  			throw new FileNotFoundException("Additional jar [" + file.getAbsolutePath() + "] does not exist");
16  		}
17  		add(file.toURI().toURL());
18  	}
19  
20  	/**
21  	 * Add the given URL to the system class loader.
22  	 *
23  	 * Note that this method uses reflection to change the visibility of the URLClassLoader.addURL() method. This will
24  	 * fail if a security manager forbids it.
25  	 *
26  	 * @param url
27  	 */
28  	public void add(URL url) {
29  		URLClassLoader sysLoader = (URLClassLoader) ClassLoader.getSystemClassLoader();
30  		Class sysClass = URLClassLoader.class;
31  
32  		try {
33  			Method method = sysClass.getDeclaredMethod("addURL", URL.class);
34  			method.setAccessible(true);
35  			method.invoke(sysLoader, new Object[]{ url });
36  		} catch (InvocationTargetException e) {
37  			throw new JarLoadingException(e);
38  		} catch (NoSuchMethodException e) {
39  			throw new JarLoadingException(e);
40  		} catch (IllegalAccessException e) {
41  			throw new JarLoadingException(e);
42  		}
43  	}
44  
45  }
