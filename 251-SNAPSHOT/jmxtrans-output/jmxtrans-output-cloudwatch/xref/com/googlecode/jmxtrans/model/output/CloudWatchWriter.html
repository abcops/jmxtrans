29  /**
30   * Writes data to <a href="http://aws.amazon.com/cloudwatch/">AWS CloudWatch</a> using the AWS Java SDK
31   *
32   * @author <a href="mailto:sascha.moellering@gmail.com">Sascha Moellering</a>
33   */
34  public class CloudWatchWriter extends BaseOutputWriter {
35  
36  	private static final Logger log = LoggerFactory.getLogger(CloudWatchWriter.class);
37  
38  	private AmazonCloudWatchClient cloudWatchClient;
39  	private String namespace;
40  
41  	private final ObjectToDouble toDoubleConverter = new ObjectToDouble();
42  
43  	@JsonCreator
44  	public CloudWatchWriter(
45  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
46  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
47  			@JsonProperty("debug") Boolean debugEnabled,
48  			@JsonProperty("namespace") String namespace,
49  			@JsonProperty("settings") Map<String, Object> settings) {
50  		super(typeNames, booleanAsNumber, debugEnabled, settings);
51  		this.namespace = MoreObjects.firstNonNull(namespace, (String) getSettings().get("namespace"));
52  		if (isNullOrEmpty(this.namespace)) throw new IllegalArgumentException("namespace cannot be null or empty");
53  	}
54  
55  	@VisibleForTesting
56  	void setCloudWatchClient(AmazonCloudWatchClient cloudWatchClient) {
57  		this.cloudWatchClient = cloudWatchClient;
58  	}
59  
60  	@Override
61  	public void validateSetup(Server server, Query query) throws ValidationException {
62  		if (cloudWatchClient == null) {
63  
64  			// Configuring the CloudWatch client
65  			// Credentials are loaded from the Amazon EC2 Instance Metadata Service
66  
67  			cloudWatchClient = new AmazonCloudWatchClient(new InstanceProfileCredentialsProvider());
68  			Region awsRegion = Regions.getCurrentRegion();
69  			if (awsRegion == null) throw new ValidationException("Problems getting metadata", query);
70  			cloudWatchClient.setRegion(awsRegion);
71  		}
72  	}
73  
74  	@Override
75  	protected void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
76  		PutMetricDataRequest metricDataRequest = new PutMetricDataRequest();
77  		metricDataRequest.setNamespace(namespace);
78  		List<MetricDatum> metricDatumList = new ArrayList<MetricDatum>();
79  
80  		// Iterating through the list of query results
81  
82  		for (Result result : results) {
83  			Map<String, Object> resultValues = result.getValues();
84  			if (resultValues != null) {
85  				for (Map.Entry<String, Object> values : resultValues.entrySet()) {
86  					try {
87  						metricDatumList.add(processResult(result, values));
88  					} catch (IllegalArgumentException iae) {
89  						log.error("Could not convert result to double", iae);
90  					}
91  				}
92  			}
93  		}
94  
95  		metricDataRequest.setMetricData(metricDatumList);
96  		cloudWatchClient.putMetricData(metricDataRequest);
97  	}
98  
99  	private MetricDatum processResult(Result result, Map.Entry<String, Object> values) {
100 		// Sometimes the attribute name and the key of the value are the same
101 		MetricDatum metricDatum = new MetricDatum();
102 		if (result.getAttributeName().equals(values.getKey())) {
103 			metricDatum.setMetricName(result.getAttributeName());
104 		} else {
105 			metricDatum.setMetricName(result.getAttributeName() + "_" + values.getKey());
106 		}
107 
108 		// Converts the Objects to Double-values for CloudWatch
109 		metricDatum.setValue(toDoubleConverter.apply(values.getValue()));
110 		metricDatum.setTimestamp(new Date());
111 		return metricDatum;
112 	}
