21  /**
22   * A writer for Log4J. It may be a nice way to send JMX metrics to Logstash for example. <br /> <br />
23   * <p/>
24   * Here is an example of MDC variables that are set by this class. <br> <br> server: localhost_9003 <br /> metric:
25   * sun_management_MemoryImpl.HeapMemoryUsage_committed <br /> value: 1251999744 <br /> resultAlias: myHeapMemoryUsage
26   * <br /> attributeName: HeapMemoryUsage <br /> key: committed <br /> epoch: 1388343325728 <br />
27   *
28   * @author Yannick Robin
29   */
30  public class Log4JWriter extends BaseOutputWriter {
31  	private final Logger log;
32  	private final String logger;
33  
34  	@JsonCreator
35  	public Log4JWriter(
36  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
37  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
38  			@JsonProperty("debug") Boolean debugEnabled,
39  			@JsonProperty("logger") String logger,
40  			@JsonProperty("settings") Map<String, Object> settings) {
41  		super(typeNames, booleanAsNumber, debugEnabled, settings);
42  		this.logger = firstNonNull(logger, (String) getSettings().get("logger"), "Log4JWriter");
43  		this.log = Logger.getLogger("Log4JWriter." + this.logger);
44  	}
45  
46  	public void validateSetup(Server server, final Query query) throws ValidationException {}
47  
48  	/**
49  	 * Set the log context and log
50  	 */
51  	public void internalWrite(Server server, final Query query, ImmutableList<Result> results) throws Exception {
52  		final List<String> typeNames = getTypeNames();
53  
54  		for (final Result result : results) {
55  			final Map<String, Object> resultValues = result.getValues();
56  			if (resultValues != null) {
57  				for (final Entry<String, Object> values : resultValues.entrySet()) {
58  					if (NumberUtils.isNumeric(values.getValue())) {
59  						String alias;
60  						if (server.getAlias() != null) {
61  							alias = server.getAlias();
62  						} else {
63  							alias = server.getHost() + "_" + server.getPort();
64  							alias = StringUtils.cleanupStr(alias);
65  						}
66  
67  						MDC.put("server", alias);
68  						MDC.put("metric", KeyUtils.getKeyString(server, query, result, values, typeNames, null));
69  						MDC.put("value", values.getValue());
70  						if (result.getKeyAlias() != null) {
71  							MDC.put("resultAlias", result.getKeyAlias());
72  						}
73  						MDC.put("attributeName", result.getAttributeName());
74  						MDC.put("key", values.getKey());
75  						MDC.put("Epoch", String.valueOf(result.getEpoch()));
76  						log.info("");
77  					}
78  				}
79  			}
80  		}
81  	}
82  
83  	public String getLogger() {
84  		return logger;
85  	}
