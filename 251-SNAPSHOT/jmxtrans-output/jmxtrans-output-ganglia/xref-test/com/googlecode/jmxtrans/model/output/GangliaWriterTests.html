17  public class GangliaWriterTests {
18  
19      /** Test validation when no parameter is set. */
20      @Test(expected = NullPointerException.class)
21      public void testValidationWithoutSettings() throws ValidationException {
22  		GangliaWriter.builder().build();
23      }
24  
25      /** Test validation when only required parameters are set. */
26      @Test
27      public void testValidationMinimalSettings() throws ValidationException {
28  		GangliaWriter writer = GangliaWriter.builder().setHost("192.168.1.144").build();
29          Query test = Query.builder()
30  				.setObj("test")
31  				.build();
32  		Server server = Server.builder().setHost("localhost").setPort("123").build();
33  		writer.validateSetup(server, test);
34          Assert.assertEquals("192.168.1.144", writer.getHost());
35          Assert.assertEquals(GangliaWriter.DEFAULT_PORT, writer.getPort());
36          Assert.assertEquals(GangliaWriter.DEFAULT_ADDRESSING_MODE.name(), writer.getAddressingMode());
37          Assert.assertEquals(GangliaWriter.DEFAULT_TTL, writer.getTtl());
38          Assert.assertEquals(GangliaWriter.DEFAULT_V31, writer.isV31());
39          Assert.assertEquals(GangliaWriter.DEFAULT_UNITS, writer.getUnits());
40          Assert.assertEquals(GangliaWriter.DEFAULT_SLOPE, writer.getSlope());
41          Assert.assertEquals(GangliaWriter.DEFAULT_TMAX, writer.getTmax());
42          Assert.assertEquals(GangliaWriter.DEFAULT_DMAX, writer.getDmax());
43          Assert.assertEquals(GangliaWriter.DEFAULT_GROUP_NAME, writer.getGroupName());
44      }
45  
46      /** Test validation when all parameters are set. */
47      @Test
48      public void testValidationAllSettings() throws ValidationException {
49  		GangliaWriter writer = GangliaWriter.builder()
50  				.setHost("192.168.1.144")
51  				.setPort(25654)
52  				.setAddressingMode("MULTICAST")
53  				.setTtl(4)
54  				.setV31(false)
55  				.setUnits("km/h")
56  				.setSlope("NEGATIVE")
57  				.setTmax(354)
58  				.setDmax(24)
59  				.setGroupName("dummy")
60  				.build();
61  
62          Query test = Query.builder()
63  				.setObj("test")
64  				.build();
65  		Server server = Server.builder().setHost("localhost").setPort("123").build();
66  		writer.validateSetup(server, test);
67          Assert.assertEquals("192.168.1.144", writer.getHost());
68          Assert.assertEquals(25654, writer.getPort());
69          Assert.assertEquals(GMetric.UDPAddressingMode.MULTICAST.name(), writer.getAddressingMode());
70          Assert.assertEquals(4, writer.getTtl());
71          Assert.assertEquals(false, writer.isV31());
72          Assert.assertEquals("km/h", writer.getUnits());
73          Assert.assertEquals(GMetricSlope.NEGATIVE, writer.getSlope());
74          Assert.assertEquals(354, writer.getTmax());
75          Assert.assertEquals(24, writer.getDmax());
76          Assert.assertEquals("dummy", writer.getGroupName());
77      }
