84  	@JsonCreator
85  	public GangliaWriter(
86  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
87  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
88  			@JsonProperty("debug") Boolean debugEnabled,
89  			@JsonProperty("host") String host,
90  			@JsonProperty("port") Integer port,
91  			@JsonProperty("addressingMode") String addressingMode,
92  			@JsonProperty("ttl") Integer ttl,
93  			@JsonProperty("v31") Boolean v31,
94  			@JsonProperty("units") String units,
95  			@JsonProperty("slope") String slope,
96  			@JsonProperty("tmax") Integer tmax,
97  			@JsonProperty("dmax") Integer dmax,
98  			@JsonProperty("groupName") String groupName,
99  			@JsonProperty("settings") Map<String, Object> settings) {
100 		super(typeNames, booleanAsNumber, debugEnabled, settings);
101 		this.host = MoreObjects.firstNonNull(host, (String) getSettings().get(HOST));
102 		this.port = MoreObjects.firstNonNull(
103 				port,
104 				Settings.getIntSetting(getSettings(), PORT, DEFAULT_PORT));
105 		this.addressingMode = computeAddressingMode(firstNonNull(
106 				addressingMode,
107 				(String) getSettings().get(ADDRESSING_MODE),
108 				""
109 		), this.host);
110 		this.ttl = MoreObjects.firstNonNull(ttl, Settings.getIntegerSetting(getSettings(), TTL, DEFAULT_TTL));
111 		this.v31 = MoreObjects.firstNonNull(v31, Settings.getBooleanSetting(getSettings(), V31, DEFAULT_V31));
112 		this.units = firstNonNull(units, (String) getSettings().get(UNITS), DEFAULT_UNITS);
113 		this.slope = GMetricSlope.valueOf(firstNonNull(
114 				slope,
115 				(String) getSettings().get(SLOPE),
116 				DEFAULT_SLOPE.name()
117 		));
118 		this.tmax = MoreObjects.firstNonNull(tmax, Settings.getIntegerSetting(getSettings(), TMAX, DEFAULT_TMAX));
119 		this.dmax = MoreObjects.firstNonNull(dmax, Settings.getIntegerSetting(getSettings(), DMAX, DEFAULT_DMAX));
120 		this.groupName = firstNonNull(
121 				groupName,
122 				(String) getSettings().get(GROUP_NAME),
123 				DEFAULT_GROUP_NAME
124 		);
125 	}

132 		// Determine the spoofed hostname

139 				TTL + ": " + ttl + ", " +

143 				TMAX + ": " + tmax + ", " +
144 				DMAX + ": " + dmax + ", " +
145 				SPOOF_NAME + ": " + spoofedHostName + ", " +
146 				GROUP_NAME + ": '" + groupName + "']");

149 	private UDPAddressingMode computeAddressingMode(String mode, String host) {
150 		// Parse and validate addressing mode setting
151 		try {
152 			return UDPAddressingMode.valueOf(mode);
153 		} catch (IllegalArgumentException iae) {
154 			try {
155 				return UDPAddressingMode.getModeForAddress(host);
156 			} catch (UnknownHostException uhe) {
157 				return DEFAULT_ADDRESSING_MODE;
158 			} catch (IOException ioe) {
159 				return DEFAULT_ADDRESSING_MODE;
160 			}
161 		}
162 	}

164 	/**
165 	 * Send query result values to Ganglia.
166 	 */
167 	@Override
168 	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
169 		for (final Result result : results) {
170 			if (result.getValues() != null) {
171 				for (final Map.Entry<String, Object> resultValue : result.getValues().entrySet()) {
172 					final String name = KeyUtils.getKeyString(query, result, resultValue, getTypeNames());
173 
174 					Object transformedValue = valueTransformer.apply(resultValue.getValue());
175 
176 					GMetricType dataType = getType(resultValue.getValue());
177 					log.debug("Sending Ganglia metric {}={} [type={}]", name, transformedValue, dataType);
178 					new GMetric(host, port, addressingMode, ttl, v31, null, spoofedHostName)
179 							.announce(name, transformedValue.toString(), dataType, units, slope, tmax, dmax, groupName);
180 				}
181 			}
182 		}
183 	}

185 	/**
186 	 * Determines the spoofed host name to be used when emitting metrics to a
187 	 * gmond process. Spoofed host names are of the form IP:hostname.
188 	 *
189 	 * @param host  the host of the gmond (ganglia monitor) to which we are
190 	 *              connecting, not null
191 	 * @param alias the custom alias supplied, may be null
192 	 * @return the host name to use when emitting metrics, in the form of
193 	 * IP:hostname
194 	 */
195 	public static String getSpoofedHostName(String host, String alias) {
196 		// Determine the host name to use as the spoofed host name, this should
197 		// be of the format IP:hostname
198 		String spoofed = host;
199 		if (StringUtils.isNotEmpty(alias)) {
200 			// If the alias was supplied in the appropriate format, use it
201 			// directly
202 			Matcher hostIpMatcher = PATTERN_HOST_IP.matcher(alias);
203 			if (hostIpMatcher.matches())
204 				return alias;
205 			// Otherwise, use the alias as the host
206 			spoofed = alias;
207 		}
208 		// Attempt to find the IP of the given host (this may be an aliased
209 		// host)
210 		try {
211 			return InetAddress.getByName(spoofed).getHostAddress() + ":" + spoofed;
212 		} catch (UnknownHostException e) {
213 			// ignore failure to resolve spoofed host
214 		}
215 		// Attempt to return the local host IP with the spoofed host name
216 		try {
217 			return InetAddress.getLocalHost().getHostAddress() + ":" + spoofed;
218 		} catch (UnknownHostException e) {
219 			// ignore failure to resolve spoofed host
220 		}
221 		// We failed to resolve the spoofed host or our local host, return "x"
222 		// for the IP
223 		return "x:" + spoofed;
224 	}

226 	/**
227 	 * Guess the Ganglia gmetric type to use for a given object.
228 	 *
229 	 * @param obj the object to inspect
230 	 * @return an appropriate {@link GMetricType}, {@link GMetricType#STRING} by default
231 	 */
232 	private static GMetricType getType(final Object obj) {
233 
234 		// FIXME This is far from covering all cases.
235 		// FIXME Wasteful use of high capacity types (eg Short => INT32)
236 
237 		// Direct mapping when possible
238 		if (obj instanceof Long || obj instanceof Integer || obj instanceof Byte || obj instanceof Short)
239 			return GMetricType.INT32;
240 		if (obj instanceof Float)
241 			return GMetricType.FLOAT;
242 		if (obj instanceof Double)
243 			return GMetricType.DOUBLE;
244 
245 		// Convert to double or int if possible
246 		try {
247 			Double.parseDouble(obj.toString());
248 			return GMetricType.DOUBLE;
249 		} catch (NumberFormatException e) {
250 			// Not a double
251 		}
252 		try {
253 			Integer.parseInt(obj.toString());
254 			return GMetricType.UINT32;
255 		} catch (NumberFormatException e) {
256 			// Not an int
257 		}
258 
259 		return GMetricType.STRING;
260 	}
