29  /**
30   * This low latency and thread safe output writer sends data to a host/port combination
31   * in the Graphite format.
32   *
33   * @see <a href="http://graphite.wikidot.com/getting-your-data-into-graphite">Getting your data into Graphite</a>
34   *
35   * @author jon
36   */
37  @NotThreadSafe
38  public class GraphiteWriter extends BaseOutputWriter {
39  	private static final Logger log = LoggerFactory.getLogger(GraphiteWriter.class);
40  
41  	private static final String DEFAULT_ROOT_PREFIX = "servers";
42  
43  	private GenericKeyedObjectPool<InetSocketAddress, Socket> pool;
44  
45  	private final String rootPrefix;
46  	private final InetSocketAddress address;
47  
48  	@JsonCreator
49  	public GraphiteWriter(
50  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
51  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
52  			@JsonProperty("debug") Boolean debugEnabled,
53  			@JsonProperty("rootPrefix") String rootPrefix,
54  			@JsonProperty("host") String host,
55  			@JsonProperty("port") Integer port,
56  			@JsonProperty("settings") Map<String, Object> settings) {
57  		super(typeNames, booleanAsNumber, debugEnabled, settings);
58  		this.rootPrefix = resolveProps(
59  				firstNonNull(
60  						rootPrefix,
61  						(String) getSettings().get("rootPrefix"),
62  						DEFAULT_ROOT_PREFIX));
63  		host = resolveProps(host);
64  		if (host == null) {
65  			host = (String) getSettings().get(HOST);
66  		}
67  		if (host == null) {
68  			throw new NullPointerException("Host cannot be null.");
69  		}
70  		if (port == null) {
71  			port = Settings.getIntegerSetting(getSettings(), PORT, null);
72  		}
73  		if (port == null) {
74  			throw new NullPointerException("Port cannot be null.");
75  		}
76  		this.address = new InetSocketAddress(host, port);
77  	}
78  
79  	public void validateSetup(Server server, Query query) throws ValidationException {
80  	}
81  
82  	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
83  		Socket socket = null;
84  		PrintWriter writer = null;
85  
86  		try {
87  			socket = pool.borrowObject(address);
88  			writer = new PrintWriter(new OutputStreamWriter(socket.getOutputStream(), UTF_8), true);
89  
90  			List<String> typeNames = this.getTypeNames();
91  
92  			for (Result result : results) {
93  				log.debug("Query result: {}", result);
94  				Map<String, Object> resultValues = result.getValues();
95  				if (resultValues != null) {
96  					for (Entry<String, Object> values : resultValues.entrySet()) {
97  						Object value = values.getValue();
98  						if (NumberUtils.isNumeric(value)) {
99  
100 							String line = KeyUtils.getKeyString(server, query, result, values, typeNames, rootPrefix)
101 									.replaceAll("[()]", "_") + " " + value.toString() + " "
102 									+ result.getEpoch() / 1000 + "\n";
103 							log.debug("Graphite Message: {}", line);
104 							writer.write(line);
105 						} else {
106 							log.warn("Unable to submit non-numeric value to Graphite: [{}] from result [{}]", value, result);
107 						}
108 					}
109 				}
110 			}
111 		} finally {
112 			if (writer != null && writer.checkError()) {
113 				log.error("Error writing to Graphite, clearing Graphite socket pool");
114 				pool.invalidateObject(address, socket);
115 			} else {
116 				pool.returnObject(address, socket);
117 			}
118 		}
119 	}
